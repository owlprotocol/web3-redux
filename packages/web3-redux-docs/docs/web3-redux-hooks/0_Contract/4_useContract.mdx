# useContract
```tsx title="src/index.tsx"
import React from 'react';
import ReactDOM from 'react-dom';
import { Provider } from 'react-redux';
import { store } from '@owlprotocol/web3-redux';

import App from './App';

ReactDOM.render(
    <React.StrictMode>
        // highlight-start
        <Provider store={store}>
            <App />
        </Provider>
        // highlight-end
    </React.StrictMode>,
);
```
